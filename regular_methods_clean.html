desc = f"""
# measurement model
automaticity =~ {' + '.join(aut)}
attitude =~ {' + '.join(att)}
norms =~ {' + '.join(nor)}
risk =~ {' + '.join(ris)}
effective =~ {' + '.join(eff)}

# regression
{target} =~ automaticity + attitude + norms + risk + effective

# residual correlations
{aut_corrs}
{att_corrs}
{nor_corrs}
{ris_corrs}
{eff_corrs}

"""
